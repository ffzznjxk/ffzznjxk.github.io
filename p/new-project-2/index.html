        /// <summary>
        /// 提示选择点
        /// </summary>
        [CommandMethod("PromptSelectPoint")]
        public void PromptSelectPoint()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取点提示
            var ppo = new PromptPointOptions("\n指定起点");
            //获取点结果
            var ppr = ed.GetPoint(ppo);
            //结果状态为OK
            if (ppr.Status == PromptStatus.OK)
            {
                //起点
                Point3d startPt = ppr.Value;
                ppo = new PromptPointOptions("\n指定终点");
                ppr = ed.GetPoint(ppo);
                if (ppr.Status == PromptStatus.OK)
                {
                    //终点
                    Point3d endPt = ppr.Value;
                    //直线
                    Line line = new Line(startPt, endPt);
                    //图形数据库
                    Database db = HostApplicationServices.WorkingDatabase;
                    //添加直线
                    db.AddNewEntity(line);
                }
                else
                {
                    ed.WriteMessage("\n取消画线");
                }
            }
            else
            {
                ed.WriteMessage("\n取消画线");
            }
        }

        /// <summary>
        /// 连续绘制直线
        /// </summary>
        [CommandMethod("AddLineMultiple")]
        public void AddLineMultiple()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取起点
            var ppr = ed.GetPoint(new PromptPointOptions("\n指定起点"));
            //结果状态为OK
            if (ppr.Status == PromptStatus.OK)
            {
                //起点
                var startPt = ppr.Value;
                //第一直线终点
                ppr = ed.GetPoint(new PromptPointOptions("\n指定下一点"));
                //图形数据库
                Database db = HostApplicationServices.WorkingDatabase;
                //结果状态为OK
                while (ppr.Status == PromptStatus.OK)
                {
                    //终点
                    Point3d endPt = ppr.Value;
                    //直线
                    Line line = new Line(startPt, endPt);
                    //添加直线
                    db.AddNewEntity(line);
                    //下一直线起点
                    startPt = endPt;
                    //获取下一直线终点
                    ppr = ed.GetPoint(new PromptPointOptions("\n指定下一点"));
                }
            }
            else
            {
                ed.WriteMessage("\n取消画线");
            }
        }

        /// <summary>
        /// 提示选择点
        /// </summary>
        [CommandMethod("DrawPolyLine")]
        public void DrawPolyLine()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取起点
            var ppr = ed.GetPoint(new PromptPointOptions("\n指定点"));
            //点列表
            List<Point3d> pts = new List<Point3d>();
            //结果状态为OK
            while (ppr.Status == PromptStatus.OK)
            {
                //添加点
                pts.Add(ppr.Value);
                //获取下一点
                ppr = ed.GetPoint(new PromptPointOptions("\n指定下一点"));
            }
            //点数大于1
            if (pts.Count > 1)
            {
                //多线
                Polyline pl = new Polyline();
                //将点加入多线
                for (int i = 0; i < pts.Count; i++)
                {
                    Point3d pt = pts[i];
                    pl.AddVertexAt(i, new Point2d(pt.X, pt.Y), 0, 0, 0);
                }
                //图形数据库
                Database db = HostApplicationServices.WorkingDatabase;
                //添加多线
                db.AddNewEntity(pl);
            }
            else
            {
                ed.WriteMessage("\n获得的点少于2个，无法绘制。");
            }
        }

        /// <summary>
        /// 绘制平行四边形
        /// </summary>
        [CommandMethod("DrawParallelogram")]
        public void DrawParallelogram()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取对象提示
            var peo = new PromptEntityOptions("\n选择第一条直线");
            //拒绝信息
            peo.SetRejectMessage("选择直线");
            //只允许直线
            peo.AddAllowedClass(typeof(Line), true);
            //获取对象结果
            var per = ed.GetEntity(peo);
            //结果状态为OK
            if (per.Status == PromptStatus.OK)
            {
                //图形数据库
                Database db = HostApplicationServices.WorkingDatabase;
                using (var tr = db.TransactionManager.StartTransaction())
                {
                    //直线1
                    var line1 = (Line)per.ObjectId.GetObject(OpenMode.ForRead);
                    //长度大于0
                    if (line1 != null && line1.Length > 0)
                    {
                        //获取点
                        var ppr = ed.GetPoint(new PromptPointOptions("\n指定点"));
                        if (ppr.Status == PromptStatus.OK)
                        {
                            var pt = ppr.Value;
                            //平行四边形点
                            List<Point3d> pts = new List<Point3d>
                            {
                                line1.StartPoint, line1.EndPoint, pt, pt - line1.Delta
                            };
                            //创建多边形
                            db.CreatePolyline(pts);
                            //删除原线
                            line1.UpgradeOpen();
                            line1.Erase();
                        }
                    }
                    else
                    {
                        ed.WriteMessage("\n直线长度过短。");
                    }
                    tr.Commit();
                }
            }
        }

        /// <summary>
        /// 绘制矩形
        /// </summary>
        [CommandMethod("DrawRectangle")]
        public void DrawRectangle()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取第一点
            var ppr = ed.GetPoint(new PromptPointOptions("\n指定点"));
            if (ppr.Status == PromptStatus.OK)
            {
                var pt1 = ppr.Value;
                //获取角点
                ppr = ed.GetCorner(new PromptPointOptions("\n指定角点"));
                if (ppr.Status == PromptStatus.OK)
                {
                    var pt2 = ppr.Value;
                    //矩形最大、最小点
                    Point2d maxPt = new Point2d(Math.Max(pt1.X, pt2.X), Math.Max(pt1.Y, pt2.Y));
                    Point2d minPt = new Point2d(Math.Min(pt1.X, pt2.X), Math.Min(pt1.Y, pt2.Y));
                    //边界点
                    List<Point2d> pts2d = new List<Point2d>
                    {
                        minPt, new Point2d(maxPt.X, minPt.Y), maxPt, new Point2d(minPt.X, maxPt.Y),
                    };
                    Database db = HostApplicationServices.WorkingDatabase;
                    //绘制矩形
                    db.CreatePolyline(pts2d);
                }
            }
        }

        /// <summary>
        /// 绘制正多边形
        /// </summary>
        [CommandMethod("DrawRegularPolygon")]
        public void DrawRegularPolygon()
        {
            //当前文档
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            //获得编辑器
            Editor ed = doc.Editor;
            //获取点
            var ppr = ed.GetPoint(new PromptPointOptions("\n指定正多边形中心"));
            if (ppr.Status != PromptStatus.OK)
                return;
            //获取第二点
            var pt1 = ppr.Value;
            ppr = ed.GetPoint(new PromptPointOptions("\n指定正多边形角点"));
            if (ppr.Status != PromptStatus.OK)
                return;
            var pt2 = ppr.Value;
            //长度
            var distance = pt2.DistanceTo(pt1);
            if (distance == 0)
                return;
            //多边形数量
            var pio = new PromptIntegerOptions("\n输入多边形边数");
            var pir = ed.GetInteger(pio);
            if (pir.Status != PromptStatus.OK)
                return;
            int sideNum = pir.Value;
            //变数不小于3
            if (sideNum < 3)
                return;
            List<Point2d> pts2d = new List<Point2d>();
            //初始角度
            var angle = (pt2 - pt1).GetAngleTo(Vector3d.XAxis);
            //负角度
            if (pt2.Y < pt1.Y)
                angle = -angle;
            //初始点
            for (int i = 0; i < sideNum; i++)
            {
                //相对向量
                var vector2d = new Vector2d(Math.Cos(angle), Math.Sin(angle)) * distance;
                //计算点
                var pt = pt1.Convert2d() + vector2d;
                pts2d.Add(pt);
                //角度增量
                angle += Math.PI * 2 / sideNum;
            }
            Database db = HostApplicationServices.WorkingDatabase;
            //添加多边形
            db.CreatePolyline(pts2d);
        }

        /// <summary>
        /// 绘制多线段
        /// </summary>
        /// <param name="db">图形数据库</param>
        /// <param name="pts2d">2维点集合</param>
        /// <param name="closed">闭合</param>
        public static ObjectId CreatePolyline(this Database db, List<Point2d> pts2d, bool closed = true)
        {
            //点数过少返回空
            if (pts2d.Count < 2)
                return ObjectId.Null;
            //多线
            Polyline pl = new Polyline();
            //将点加入多线
            for (int i = 0; i < pts2d.Count; i++)
                pl.AddVertexAt(i, pts2d[i], 0, 0, 0);
            //闭合
            pl.Closed = closed;
            //添加多线
            return db.AddNewEntity(pl)[0];
        }

        /// <summary>
        /// 绘制多线段
        /// </summary>
        /// <param name="db">图形数据库</param>
        /// <param name="pts2d">3d点集合</param>
        /// <param name="closed">闭合</param>
        public static ObjectId CreatePolyline(this Database db, List<Point3d> pts, bool closed = true)
        {
            //2d点集合
            var pts2d = new List<Point2d>();
            for (int i = 0; i < pts.Count; i++)
                pts2d.Add(pts[i].Convert2d());
            //var pts2d = pts.Select(p => new Point2d(p.X, p.Y)).ToList();
            return db.CreatePolyline(pts2d, closed);
        }

        /// <summary>
        /// 3d坐标转2d
        /// </summary>
        /// <param name="pt"></param>
        /// <returns></returns>
        public static Point2d Convert2d(this Point3d pt)
        {
            Document doc = AcadApp.DocumentManager.MdiActiveDocument;
            Editor ed = doc.Editor;
            return pt.Convert2d(ed.GetPlane());
        }

        /// <summary>
        /// 获取法向量平面
        /// </summary>
        /// <param name="ed"></param>
        /// <returns></returns>
        public static Plane GetPlane(this Editor ed)
        {
            return new Plane(Point3d.Origin, ed.GetNormal());
        }

        /// <summary>
        /// 当前坐标系法向量
        /// </summary>
        /// <param name="ed"></param>
        /// <returns></returns>
        public static Vector3d GetNormal(this Editor ed)
        {
            return Vector3d.ZAxis.TransformBy(ed.CurrentUserCoordinateSystem);
        }
